jQuery(document).ready(function($) {
	var totalPages = 60;
	var perPage = 18;
	var url = 'http://www.dexerto.com/';
	var $archiveList = $('#archive-content');
		
	var doingAjax = false;
	var lastLoadedPage = 0; 
	var page = 2; // First request == page 2
		
	$('body').on('click','.load-more',function(e){
		if  ( (totalPages > 1 && !doingAjax) && (lastLoadedPage !== totalPages) ){
			loadArticle(page);
			
		}
	});
	
	function loadArticle(pageNumber)
	{
		doingAjax = true;  
		lastLoadedPage = page;
		var $btn = $('.load-more');
		$btn.addClass('loading');  
		$.ajax({
			url: url+'?page='+pageNumber,
			type:'GET',
			dataType:'html',
			success: function(html){
				var items = $(html).find('#archive-content').html();
				$('body').trigger( "archive:paged", [items] );
				
				if (page == totalPages) {
					$btn.remove();
					$(document.body).trigger("sticky_kit:recalc");
				} else {
					$btn.removeClass('loading');
				}
				page++;
				doingAjax = false;
			}
		});
		return false;
	}
});
