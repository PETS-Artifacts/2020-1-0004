  // this function calculates the positions of each li and each ul inside the variation area
  function placeVariationItems(jqSelector,anchorWrap,direction) {
    var colOffset = 0;     // col-offset is the return value of the complete left or right variation area
    var prevColWidth = 0;  // the previous col-width is the width of the previous ul to place the current ul at the right position
    
    jq('ul',jqSelector).each( function() {
      var $col = jq(this);
      var colWidth = $col.outerWidth();
      var wBorder = (parseInt($col.css('border-left-width')) + parseInt($col.css('border-right-width')));
      var items = jq('li',$col);
      var itemCount = items.length;
      
      var currentHeight = 0;
      var finalColWidth = colWidth;
      var itemTop = 0;
      var itemLeft = 0;
      jq(items).each(function(){
        jq(this).css({
          'left':itemLeft,
          'top':itemTop,
          'position':'absolute',
          'width':(colWidth - wBorder)
        });

        itemTop = itemTop + jq(this).outerHeight();

        if(itemTop > anchorWrap.height()) {
          finalColWidth = finalColWidth + colWidth;
          itemLeft = itemLeft + colWidth;
          itemTop = 0;
          
          jq(this).css({
            'left':itemLeft,
            'top':itemTop,
          });
          
          itemTop = itemTop + jq(this).outerHeight();
        }
      });
      
      $col.css({
        'width':(finalColWidth - wBorder),
        'position':'absolute',
        'left' : prevColWidth
      });
      
      prevColWidth = finalColWidth;
      
      if(direction === 'left') {
        jqSelector.css('left',0);
      }
      else {
        jqSelector.css('right',0);
      }
      
      colOffset = colOffset + finalColWidth;
    });
    
    return colOffset;
  }

  function GetPixiDeliveryPeriod(ProductID, WebPath, ProductDeliveryPeriod, ManufacturerDeliveryPeriod, StockLevelAlert, MinOrder, ShopStockLevel, AvaibilyText, ProductIsAvailable){
    //var BaseURL         = "https://www.fc-moto.de";
    var BaseURL           = "http://www.fc-moto.de";
    var AjaxURL           = BaseURL + WebPath + "/stockstate_" + ProductID + ".json?PreventCache=" + Date.now();
    var LowOnStockAction  = 0;
    
    if(ShopStockLevel > 0){ return; }
    if(ManufacturerDeliveryPeriod == ''){ return; } 
    if(ProductDeliveryPeriod == ''){ return; }
    if(AvaibilyText == ''){ return; }
    
    jQuery.ajax({      
      type: 'GET',
      async: false,
      url: AjaxURL,
      dataType: 'json',
      success: function(data) {
        if(data.InternalStock <= 0 && data.ExternalStock > 0){
          ChangeDeliveryPeriod(ProductID, ProductDeliveryPeriod, ManufacturerDeliveryPeriod, AvaibilyText);        
        }

        var TrafficLight = ChangeTrafficLight(ProductID, data.AvailableStock, StockLevelAlert, MinOrder);

        if(LowOnStockAction == 2 && TrafficLight == "ProductOutStockIcon" && ProductIsAvailable) {
          ChangeBasketButton(ProductID, 0);        
        }else{
          ChangeBasketButton(ProductID, 1);        
        }
      }
    });
  
    return;
  }
  
  function ChangeMicroDataAvailability(ProductID, ProductStock, StockLevelAlert, MinOrder, LowOnStockAction)

  }

/** 
 * show only x Elements in FilterList for Manufacturer and Size
 */
function transformFilterList() {
    var ShowAll = 'show all',
        size = 0,
        showElements = 6;

    size = jq('.RemoteSearchFacet.Manufacturer li').size();
    jq('.Middle .RemoteSearchFacetSelect.Manufacturer li.ManufacturerFilter').css({'margin-left':'0px'});
    
    if(size > showElements) {
        while (size > showElements) {
            jq(".RemoteSearchFacet.Manufacturer li:nth-child("+size+")").hide();
            size = size - 1;
        }
        jq('.RemoteSearchFacet.Manufacturer ul').append("<li class='ManufacturerFilter'><a href='#' onclick=showAllManufaturer(event)>"+ShowAll+"</a></li>");
    }

    size = jq(".RemoteSearchFacet.Size li").size(); 
    
    if(size > showElements) {
        while (size > showElements) {
            jq(".RemoteSearchFacet.Size li:nth-child("+size+")").hide();
            size = size - 1;
        }
        jq('.RemoteSearchFacet.Size ul').append("<li class='SizeFilter'><a href='#' onclick=showAllSizes(event)>"+ShowAll+"</a></li>");
    }
}

}

}
