function hideEl(element) { //HIDE ELEMENT
    if (element.id) {
        document.getElementById(element.id).classList.add('hidden');

function getStyle(el, cssprop) {
    if (el.currentStyle) { //IE
        return el.currentStyle[cssprop];
    } else if (document.defaultView && document.defaultView.getComputedStyle) { //Firefox
        return document.defaultView.getComputedStyle(el, "")[cssprop];
    } else { //try and get inline style
        return el.style[cssprop];
    }
}

function removeChildren(element) {
    //REMOVES CHILDREN OF AN ELEMENT
    while (element.hasChildNodes()) {
        element.removeChild(element.lastChild);
    }
}

function getPage(url, parent, appendTo) {
    appendTo = appendTo || '.help__answer__box';
    var xhr = new XMLHttpRequest();
    try {
        xhr.onreadystatechange = function () {
            if (xhr.readyState === 4) {
                var parser = new DOMParser(),
                    docu = parser.parseFromString(xhr.responseText, "text/html"),
                    target = docu.querySelector(parent),
                    main = docu.querySelector('#main'),
                    styles = "<style>" + main.getElementsByTagName('style')[0].textContent + "</style>";
                document.querySelector(appendTo).parentElement.innerHTML = styles + document.querySelector(appendTo).parentElement.innerHTML;
                document.querySelector(appendTo).appendChild(target);
            }
        };
        xhr.open("GET", url, false);
        xhr.send(null);
    } catch (e) {
        console.log('Error With Data Request');
    }
}

function setQuery(first, second) { //USES HTML5 PUSHSTATE TO CHANGE AND SET URLS - WILL ERROR IN DEV ENVIRONMENT
    var stater = ["home", first, second];
    if (first && !second) { //IF ONLY FIRST TOPIC EXISTS - PUSH THIS STATE
        window.history.pushState(stater, null, "/help?" + first);
        setGA("/help?" + first); //SET GA AND SEND PAGE VIEW
    } else if (first) { //IF BOTH FIRST AND SECOND TOPICS EXIST - PUSH THIS STATE
        window.history.pushState(stater, null, "/help?" + first + "&" + second);
        setGA("/help?" + first + "&" + second); //SET GA AND SEND PAGE VIEW
    }
    if (!first) { //IF NEITHER FIRST NOR SECOND EXIST THEN SET DEFAULT PAGE STATE
        window.history.pushState(stater, null, "/help");
        setGA("/help"); //SET GA AND SEND PAGE VIEW
    }
    popu = stater.filter(function (val) { //FOR POPSTATE - BACK AND FORWARD BUTTONS
        return val !== undefined;
    });
}

            cell.classList.add('helpSectionNormalBox');

            cell.classList.add('helpSectionBoxSecond');

        appendToHere.appendChild(cell);
    } else if (appendToHere === sideTopicsContainer) {
        //IF CREATING SIDE TOPICS
        listItem.appendChild(anchor);
        if (text.hasOwnProperty('link')) {
            anchor.setAttribute('href', text.link);
            anchor.setAttribute('target', '_self');
        }

        if (text.hasOwnProperty('link')) {
            anchor.setAttribute('href', text.link);
            anchor.setAttribute('target', '_self');
        }
        anchor.innerHTML = text.name;
        appendToHere.appendChild(anchor);
    }
}

function searching(term, fromSearchBar) { //MAIN SEARCHING FUNCTION - USED FOR CLICKS AND TYPING
    var searchTerm = term.toLowerCase(),
        runOnce = false;
    [helpContentArea, breadCrumbInner, contentHeaderContainer, helpErrorMessage].forEach(fadeOut);
    setTimeout(function () {
        removeChildren(helpCellsAppendTo);
        for (i = 0; i < sections.length; i += 1) { //SEARCH THROUGH EVERY SECTION
            for (r = 0; r < sections[i].parentSections.length; r += 1) { //SEARCH PARENT SECTION NAMES AND CREATES CLICKABLE CELLS IF MATCH
                if (searchTerm === sections[i].parentSections[r].toLowerCase()) {
//                    console.log(1);
                    createCells(sections[i].section, helpCellsAppendTo, true, true);
                    if (runOnce === false) {
                        breadCrumbs(searchTerm);
                        setQuery(createLink(searchTerm));
                        changePageAttributes(searchTerm);
                        for (m = 0; m < mainSections.length; m += 1) {
                            if (mainSections[m].mainSection.toLowerCase().indexOf(searchTerm) > -1) {
                                mainHeaderText(searchTerm, mainSections[m].mainSectionImg);
                                break;
                            }
                        }
                        runOnce = true;
                        if (searchTerm.indexOf('all access') >= 0) {
                            phoneChange(false, 'all access');
                        } else {
                            phoneChange(true);
                        }
                    }
                    break;
                }
            }
            if (runOnce === false) { //SEARCH THROUGH KEYWORDS AND CREATE CELLS IF KEYWORDS MATCH ANYWHERE
                for (k = 0; k < sections[i].keywords.length; k += 1) {
                    if (searchTerm === sections[i].keywords[k].toLowerCase()) {
                        createCells(sections[i].section, helpCellsAppendTo, true, true);
                        breadCrumbs(searchTerm);
                        setQuery(createLink(searchTerm));
                        mainHeaderText(searchTerm);
                        changePageAttributes(searchTerm);
                    }
                }
                if (sections[i].section.toLowerCase().indexOf(searchTerm) >= 0 && fromSearchBar === true) { //SEARCH THROUGH SECTION NAMES AND IF MATCH
                    //IF SEARCH BAR IS USED
                    createCells(sections[i].section, helpCellsAppendTo, true, true);
                    breadCrumbs("Search Results");
                    setQuery();
                    mainHeaderText("Search Results");
                    changePageAttributes("Search Results");
                } else if (sections[i].section.toLowerCase() === searchTerm && fromSearchBar !== true) { //MOSTLY ANSWERS FROM MAIN SECTION
                    createAnswers(sections[i].answer.block);
                    breadCrumbs(sections[i].parentSections[0], searchTerm);
                    setQuery(createLink(sections[i].parentSections[0]), createLink(searchTerm));
                    mainHeaderText(searchTerm);
                    changePageAttributes(searchTerm);
                    if (sections[i].hasOwnProperty('getPage')) {
                        getPage('https://www.bodybuilding.com/store/affiliate.htm', '.lpContent');
                    }
                }
                
                if (sections[i].section.toLowerCase().indexOf(searchTerm) >= 0 && fromSearchBar !== true) {
                    if (sections[i].section.toLowerCase().indexOf('all access') >= 0) {
                        phoneChange(false, 'all access');
                    } else {
                        phoneChange(true);
                    }
                }
//                    contactData(sections[i].parentSections[0].toLowerCase(), searchTerm);
            }
        }
        if (searchBar.value.length > 0 && helpCellsAppendTo.querySelectorAll('.helpSectionBoxWrap').length === 0) {
            fadeIn(helpErrorMessage);
        } 
//        else if (helpCellsAppendTo.querySelectorAll('.helpSectionBoxWrap').length === 0) {
//            reset();
//        }
        [helpContentArea, breadCrumbInner, contentHeaderContainer].forEach(fadeIn);
    }, tSpeed);
}

function initLoad() { //INTIAL LOADING
//    var searchURL = window.location.search;
    firstHeader = helpJsonData.mainSectionHeader;
    document.querySelector('.help__side__bar__header').textContent = helpJsonData.popularTopicsHeader;
    for (i = 0; i < helpJsonData.popularTopics.length; i += 1) { //CREATE SIDE BAR LINKS
        createCells(helpJsonData.popularTopics[i], sideTopicsContainer);
    }
    document.querySelector('.help__quick__links__header').textContent = helpJsonData.selfHelpLinksHeader;
    for (i = 0; i < helpJsonData.selfHelpLinkTopics.length; i += 1) { //CREATE QUICK LINKS
        createCells(helpJsonData.selfHelpLinkTopics[i], quickLinksContainer);
    }
    //LOOK AT URL FOR THINGS TO SEARCH FOR
    if (checkURL()) {
        var url = window.location;
        fullLoadQueries = window.location.search.split('?')[1];
        if (url.href.indexOf('searchTerm') >= 0) {
            var end = fullLoadQueries.split('=')[1];
            splitQueries = end.split('&')[0].replace(/%20/g, ' ');
            searching(splitQueries, true);
        } else if (url.href.indexOf('&') <= -1) {
            searching(fullLoadQueries.toString().replace(/-/g, ' '));
        } else if (url.href.indexOf('&') >= 0) {
            searching(fullLoadQueries.split('&')[1].toString().replace(/-/g, ' '));
        }
    } else { //IF THERE IS NOTHING IN THE URL TO SEARCH FOR
        contentHeader.textContent = firstHeader;
        for (i = 0; i < mainSections.length; i += 1) {
            createCells(mainSections[i].mainSection, helpCellsAppendTo, true, false, mainSections[i].mainSectionImg);
        }
    }
    mainHeaderText(firstHeader);
}

        initLoad(); //RUN INITIAL LOAD FUNCTION AGAIN AFTER RESET
		[helpContentArea, breadCrumbInner, contentHeaderContainer].forEach(fadeIn); //FADE IN RESET ELEMENTS
    }, tSpeed);
    return;
}
