		function changeLinks () {
			if (vp.activityData['openhouse'] === 'true') {
				$('a').each(function () {
					$a = $(this);
					var newLink = $a.attr('data-href-o');
					var originalLink = $a.attr('href');
					if ( newLink ) {
						$a.attr('data-original-href', originalLink);
						$a.attr('href', newLink);
					}
				});
				window.ohInterval = setInterval(restoreLinks, 1000);
			}
		}

					}
				});
